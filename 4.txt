#!/bin/bash
# Practical 4 - Set 2 (Full Log View + Process Count)

# Path to log file
LOG_FILE=~/Desktop/Log/ProcessLog

# Check if log file exists
if [ ! -f "$LOG_FILE" ]; then
    echo "Log file not found at: $LOG_FILE"
    exit 1
fi

echo "========= FULL PROCESS LOG ========="
cat "$LOG_FILE"  # Display all process logs
echo "===================================="
echo ""

echo "======== PROCESS INVOCATION SEARCH ========"
read -p "Enter the process name to search: " PROC_NAME

# Search for process and count occurrences
COUNT=$(grep -cw "$PROC_NAME" "$LOG_FILE")

if [ $COUNT -gt 0 ]; then
    echo "The process '$PROC_NAME' has been invoked $COUNT time(s)."
else
    echo "The process '$PROC_NAME' was not found in the log."
fi

    sudo less /var/log/mysql/error.log
else
    echo "MySQL log file not found at /var/log/mysql/error.log"

echo "================= KERNEL LOG ================="
if [ -f /var/log/kern.log ]; then 
    sudo less /var/log/kern.log
else
    echo "Kernel log file not found at /var/log/kern.log"
fi

echo "================= MAIL LOG ================="
if [ -f /var/log/mail.log ]; then 
    sudo less /var/log/mail.log
else
    echo "Mail log file not found at /var/log/mail.log"
fi
